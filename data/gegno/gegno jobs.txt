# Mining Jobs I

mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Gather 10 tons of aluminum"
	description "Mine and collect 10 tons of aluminum, then bring the minerals back to the Gegno at <origin>."
	source
		attributes gegno mining
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10 + "Gegno Asteroid Mining II: offered" * 10
	to accept
		"cargo space" > 10
	on visit
		dialog "You've returned to <origin>, but do not have 10 tons of aluminum in your cargo hold. Make sure to mine 10 tons of aluminum and collect it into your cargo before you land."
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno mining payment dialog"

# Mining Jobs II

mission "Gegno Asteroid Mining II - Neodymium + Copper"
	job
	repeat
	name "Gather copper and neodymium"
	description "Mine and collect 15 tons of copper and 5 tons of neodymium. Bring the minerals back to the Gegno at <origin>."
	source
		attributes gegno mining
	to offer
		has "Gegno Asteroid Mining II: offered"
		or
			not "Gegno Asteroid Mining II - Neodymium + Copper: offered"
			random < 10 + "Gegno Asteroid Mining III: offered" * 10
	to accept
		"cargo space" > 20
	on visit
		dialog "You've returned to <origin>, but do not have all of the required minerals in your cargo hold. Make sure to mine and collect 15 tons of copper and 5 tons of neodymium before you return."	
	on complete
		"gegno: mining jobs" ++
		payment 128000
		outfit "Copper" -15
		outfit "Neodymium" -5
		dialog phrase "gegno mining payment dialog"
