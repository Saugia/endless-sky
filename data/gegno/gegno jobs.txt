mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Gather 10 tons of Aluminum"
	description "Mine and collect 10 tons of Aluminum, then bring the minerals back to the Gegno at <origin>."
	source "Tschyss"
	destination "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
		"cargo space" > 5
	on visit
		dialog "You've returned to <origin>, but do not have 10 tons of Aluminum mined and in your ship. Make sure to mine 10 tons of Aluminum and collect it into your cargo before you land."
	on complete
		"gegno: mining jobs" ++
		payment 36000
		dialog phrase "gegno mining payment dialog"
		outfit "Aluminum" -10
