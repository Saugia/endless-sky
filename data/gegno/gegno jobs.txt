mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Gather 10 tons of Aluminum from asteroids"
	description "Mine and collect 10 tons of Aluminum, then bring the minerals back to the Gegno at <source>."
	source "Tschyss"
	destination "Tschyss"
	to offer
		has "Gegno Asteroid Mining Prologue: done"
		random < 10
		"cargo space" > 5
	on compete
		"gegno: mining jobs" ++
		payment 10000
		dialog phrase "gegno mining payment dialog"
		outfit "Aluminum" -10
