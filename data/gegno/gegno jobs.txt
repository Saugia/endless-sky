# Mining Jobs

mission "Gegno Asteroid Mining - Aluminum"
	job
	repeat
	name "Gather 10 tons of aluminum"
	description "Mine and collect 10 tons of aluminum, then bring the minerals back to the Gegno at <origin>."
	source
		attributes gegno mining
	to offer
		random < 20
	to accept
		"cargo space" > 10
	on visit
		dialog "You've returned to <origin>, but do not have 10 tons of aluminum in your cargo hold. Make sure to mine 10 tons of aluminum and collect it into your cargo before you land."
	on complete
		"gegno: mining jobs" ++
		payment 36000
		outfit "Aluminum" -10
		dialog phrase "gegno mining payment dialog"

# Passenger Jobs

mission "Gegno Passengers [0]"
	clearance
	name "Passenger transport to <planet>"
	job
	repeat
	description "A Gegno is paying <payment> to be taken to <destination>."
	passengers 1
	source
		government "Gegno"
	to offer
		random < 75
	to accept
		or
			"flagship model: Marl"
			has "gegno: passenger acceptance"
	destination
		distance 1 3
		government "Gegno"
	on visit
		dialog phrase "generic passenger on visit"
	on complete
		"gegno: jobs" ++
		payment
		dialog phrase "gegno passenger dropoff payment"

# Tschyss Recovery Jobs

mission "Gegno: Tschyss Workers [0]"
	name "Transport <bunks> relief workers"
	job
	repeat
	description "To help maintain the workforce mending damage to the station, transport <bunks> of Gegno workers from <stopovers> to <origin>."
	source "Tschyss"
	passengers 8 2 .3
	to offer
		has "Tschyss Recovery Jobs: offered"
		or
			not "Gegno: Tschyss Workers [0]: offered"
			random < 30
		not "event: gegno: tschyss recovery"
	stopover "Ef Aourtdye"
	on stopover
		dialog "Test pickup workers."
	on visit
		dialog phrase "generic cargo on visit"
	on complete
		payment
		"gegno: recovery jobs" ++
		dialog phrase "gegno tschyss worker support"
